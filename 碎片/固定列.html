    引用  <script type="text/javascript" src="http://static.b2b.com/js/fixcol.js"></script>
    defaults : {
		wrapClass: ".scroll",  // 包裹元素
        scrollClass: "",       // 滚动元素 ，如果包裹元素和滚动元素为同一个，可以设置为空
        autoWrap : true,//自动包裹,默认true ,则自动产生包裹元素最后列的元素 div 中
		hightlightClick: false,    //是否启用 行鼠标点击 高亮并选中，默认将选择改行中的 checkbox , 如果希望覆盖默认选择中行为，则传如自定义函数即可
        hightlightCallback:function(event){
            $("input[type='checkbox']",event.currentTarget).each(function(){
                $(this).prop("checked",!$(this).prop("checked"));
            });
        },
        width:80, // 固定列宽度

        //奇偶数 换色，一般不需要配置
        rowBgColor:{
            thead:{
               even:"#e4e4e4",
               odd:"#e4e4e4"
            },
            tbody:{
                even: "#ffffff",
                odd:"#f0f0f0"
            }
        },
        //固定列做边框色值
        borderLeft:"1px solid #E0E0E0"
	}

    调用范例:
            var f = new FixCol({wrapClass:".scroll",hightlightClick:true,width:100});
            var f2 = new FixCol({wrapClass:".scroll2",hightlightClick:true,width:100,autoWrap:true});
            var m = new FixCol({wrapClass:"#g2",scrollClass:".k-grid-content",hightlightClick:true,width:100,autoWrap:true});
